Program patito;
VARS
    int: a;

function int fibonacci(int n)
VARS
    int: n1, n2; {
    if (n <= 0) then {
        return(-1);
    }
    else {
        if (n == 1) then {
            return(1);
        }
        else {
            if (n == 2) then {
                return(1);
            }
            else {
                n1 = fibonacci(n - 1);
                n2 = fibonacci(n - 2);
                return(n1 + n2);
            }
        }
    }
}

main() {
    write(fibonacci(20));
}
