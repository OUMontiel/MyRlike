Program patito;
VARS
    int: a;

function int fibonacci(int n) {
    if (n <= 0) then {
        return(-1);
    }
    else {
        if (n == 1) then {
            return(1);
        }
        else {
            if (n == 2) then {
                return(1);
            }
            else {
                return(fibonacci(n - 1) + fibonacci(n - 2));
            }
        }
    }
}

main() {
    write(fibonacci(8));
}
