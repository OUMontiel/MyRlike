Program patito;
VARS
    int: a;

function int fibonacci(int n)
VARS
    int: i, n1, n2, result; {
    n1 = 0;
    n2 = 1;
    if (n < 0) then {
        return(-1);
    }
    else {
        if (n == 0) then {
            return(n1);
        }
        else {
            if (n == 1) then {
                return(n2);
            }
            else {
                for i = 3 to n + 1 do {
                    result = n1 + n2;
                    n1 = n2;
                    n2 = result;
                }
                return(result);
            }
        }
    }
}

main() {
    write(fibonacci(8));
}
