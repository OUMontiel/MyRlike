Program patito;
VARS
    int: a;

function int factorial(int n)
VARS
    int: i, result; {
    if (n < 0) then {
        return(-1);
    }
    else {
        if (n == 0) then {
            return(1);
        }
        else {
            return(factorial(n - 1) * n);
        }
    }
}

main() {
    write(factorial(8));
}
