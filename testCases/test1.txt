Program patito;
VARS
    int: a;
    float: b;
    char: c;

function int factorial(int n)
VARS
    int: i, result; {
    if (n <= 0) then {
        return(-1);
    }
    else {
        result = 1;
        for i = 1 to n + 1 do {
            result = result * i;
        }
        return(result);
    }
}

main() {
    write(factorial(-4));
}
